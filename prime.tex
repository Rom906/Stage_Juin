\begin{algorithm}
\DontPrintSemicolon
\KwData{$x, y$ two integers}
\KwResult{The product $x \times y$}
$r \gets 0$\;
\While{$x \neq 0$}{
  \If{$x$ is odd}{
    $x \gets x - 1$\;
    $r \gets r + y$\;
  }
  $x \gets x / 2$\;
  $y \gets y \times 2$\;
}
\Return{$r$}
\end{algorithm}
